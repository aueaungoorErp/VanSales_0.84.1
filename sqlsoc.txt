CREATE TABLE VANARLINE
   (VANARL_KEY           INTEGER NOT NULL,
    VANARL_VANCNF        INTEGER NOT NULL,
    VANARL_ARL           INTEGER NOT NULL,
    CONSTRAINT VANARLPRIMARY PRIMARY KEY (VANARL_KEY),
    CONSTRAINT VANARL2VANCNF FOREIGN KEY (VANARL_VANCNF)

    CONSTRAINT VANARL2ARL    FOREIGN KEY (VANARL_ARL)
